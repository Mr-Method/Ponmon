CREATE TABLE IF NOT EXISTS `pon_fdb` (
  `mac` char(18) NOT NULL,
  `vlan` smallint NOT NULL,
  `llid` char(16) NOT NULL,
  `olt_id` smallint NOT NULL,
  `name` varchar(32) DEFAULT NULL,
  `uid` int DEFAULT NULL,
  `time` int NOT NULL DEFAULT '0',
  PRIMARY KEY (`mac`),
  KEY `llid` (`llid`),
  KEY `name` (`name`),
  KEY `olt_id` (`olt_id`),
  KEY `vlan` (`vlan`),
  KEY `uid` (`uid`)
) ENGINE=InnoDB COMMENT='FDB cache';
